usage: relay-kit continuity auto [-h] [--json]
                                 [--phase {start,resume,before-final,handoff}]
                                 [--objective OBJECTIVE] [--lane LANE]
                                 [--blocker BLOCKER] [--next-step NEXT_STEP]
                                 [--note NOTE] [--reason REASON]
                                 [--receiver RECEIVER]
                                 [--max-age-minutes MAX_AGE_MINUTES]
                                 [--force-checkpoint] [--force-rehydrate]
                                 [project_path]

positional arguments:
  project_path          Target project root

options:
  -h, --help            show this help message and exit
  --json                Emit JSON output
  --phase {start,resume,before-final,handoff}
                        Session lifecycle phase to handle
  --objective OBJECTIVE
                        Current objective summary
  --lane LANE           Active lane id
  --blocker BLOCKER     Current blocker
  --next-step NEXT_STEP
                        Exact next step
  --note NOTE           Optional auto continuity note
  --reason REASON       Reason for handoff file naming
  --receiver RECEIVER   Intended handoff receiver
  --max-age-minutes MAX_AGE_MINUTES
                        Override policy checkpoint age threshold
  --force-checkpoint    Always write a checkpoint
  --force-rehydrate     Always rehydrate when a manifest exists
