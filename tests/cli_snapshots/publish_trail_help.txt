usage: relay-kit publish trail [-h] [--channel {pypi,testpypi,internal}]
                               [--target-version TARGET_VERSION]
                               [--dist-dir DIST_DIR]
                               [--evidence-dir EVIDENCE_DIR] [--ci-url CI_URL]
                               [--release-url RELEASE_URL]
                               [--package-url PACKAGE_URL]
                               [--shell {powershell,bash}] [--allow-dev]
                               [--output-file OUTPUT_FILE]
                               [--markdown-file MARKDOWN_FILE] [--strict]
                               [--json]
                               [project_path]

positional arguments:
  project_path          Project root to inspect

options:
  -h, --help            show this help message and exit
  --channel {pypi,testpypi,internal}
  --target-version TARGET_VERSION
                        Expected package version
  --dist-dir DIST_DIR   Distribution artifact directory (default:
                        <project>/dist)
  --evidence-dir EVIDENCE_DIR
                        Directory for captured twine/upload logs (default:
                        <project>/.tmp/relay-publication/<version>)
  --ci-url CI_URL       Remote CI evidence URL
  --release-url RELEASE_URL
                        GitHub release or release-note URL
  --package-url PACKAGE_URL
                        Package index evidence URL
  --shell {powershell,bash}
  --allow-dev           Allow dev/local versions on the selected channel
  --output-file OUTPUT_FILE
                        JSON trail output path (default: <project>/.relay-
                        kit/release/publication-trail.json)
  --markdown-file MARKDOWN_FILE
                        Markdown trail output path (default: <project>/.relay-
                        kit/release/publication-trail.md)
  --strict              Return non-zero unless the publication trail is ready
  --json                Emit machine-readable publication trail
