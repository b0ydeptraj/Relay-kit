usage: relay-kit context explain-symbol [-h] --symbol SYMBOL [--limit LIMIT]
                                        [--index-file INDEX_FILE] [--json]
                                        [project_path]

positional arguments:
  project_path          Project root to inspect

options:
  -h, --help            show this help message and exit
  --symbol SYMBOL       Symbol name to explain
  --limit LIMIT         Maximum definition/reference count
  --index-file INDEX_FILE
                        Optional context index JSON path
  --json                Emit machine-readable symbol explanation
