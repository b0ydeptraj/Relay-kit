usage: relay-kit support bundle [-h]

                                [--output-file OUTPUT_FILE]
                                [--evidence-limit EVIDENCE_LIMIT] [--json]
                                [project_path]

  project_path          Project root to inspect

  --policy-pack {baseline,enterprise,team}
                        Policy pack used for support diagnostics
  --output-file OUTPUT_FILE
                        Output path (default: <project>/.relay-
                        kit/support/support-bundle.json)
  --evidence-limit EVIDENCE_LIMIT
                        Recent evidence ledger events to include
  --json                Emit bundle payload and output path as JSON
